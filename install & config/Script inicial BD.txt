
CREATE TABLE `User` (
  `VetId` int(11) NOT NULL,
  `UserName` varchar(40) NOT NULL,
  `UserEmail` varchar(70) NOT NULL,
  `UserPass` varchar(200) NOT NULL,
  `UserLevel` char(1) not null,
  `UserPhone` varchar(30) default null,
  PRIMARY KEY (`VetId`, `UserName`)
);

CREATE TABLE `Pet` (
  `VetId` int(11) NOT NULL,
  `PetId` int(11) NOT NULL,
  `PetName` varchar(70) NOT NULL,
  `PetSpecie` varchar(70) DEFAULT NULL,
  `PetAge` int(11) DEFAULT NULL,
  `PetSize` varchar(70) NOT NULL,
  `PetWeight` int(11) DEFAULT NULL,
  PRIMARY KEY (`VetId`,`PetId`)
) ;

CREATE TABLE `Payments` (
  `VetId` int(11) NOT NULL,
  `PaymentId` int(11) NOT NULL,
  `ClientId` int(11) NOT NULL,
  `PaymentPrice` int(11) NOT NULL,
  `PaymentAMount` int(11) NOT NULL,
  `PaymentDate` date DEFAULT NULL,
  `PaymentDesc` varchar(70) DEFAULT NULL,
  PRIMARY KEY (`VetId`,`PaymentId`)
);


CREATE TABLE `Matters` (
  `VetId` int(11) NOT NULL,
  `MatterId` int(11) NOT NULL,
  `MatterDesc` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`VetId`, `MatterId`)
);

CREATE TABLE `Dates` (
  `VetId` int(11) NOT NULL,
  `DateId` int(11) NOT NULL,
  `DateDay` date NOT NULL,
  `DateHourStart` varchar(15) NOT NULL,
  `DateHourEnd` varchar(15) NOT NULL,
  `PetId` int(11) DEFAULT NULL,
  `ClientId` int(11) DEFAULT NULL,
  `DateMatter` int(11) DEFAULT NULL,
  `DateAddress` varchar(70) DEFAULT NULL,
  PRIMARY KEY (`VetId`,`DateDay`,`DateHourStart`,`DateHourEnd`)
);


CREATE TABLE `Veterinary` (
  `VetId` int(11) NOT NULL AUTO_INCREMENT,
  `VetName` varchar(100) NOT NULL,
  PRIMARY KEY (`VetId`)
);



CREATE TABLE `ClientPet` (
  `VetId` int(11) NOT NULL,
  `ClientId` int(11) NOT NULL,
  `PetId` int(11) NOT NULL,
  PRIMARY KEY (`VetId`,`ClientId`,`PetId`)
) ;


CREATE TABLE `Client` (
  `VetId` int(11) NOT NULL,
  `ClientId` varchar(25) NOT NULL,
  `ClientName` varchar(70) NOT NULL,
  `ClientMail` varchar(100) DEFAULT NULL,
  `ClientPhone` varchar(30) DEFAULT NULL,
  `ClientBDate` date DEFAULT NULL,
  `ClientAddress` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`VetId`,`ClientId`)
);

CREATE TABLE `Jail` (
	`VetId`         int(11) NOT NULL,
	`JailId`        int(11) NOT NULL,
	`JailLarge`     int(11) NOT NULL,
	`JailHeight`    int(11) NOT NULL,
	`JailOcupation` char(1) NOT NULL,
	`JailState`		int(1)  NOT NUll,
	PRIMARY KEY (`VetId`, `JailId`)
);

CREATE TABLE `PetHistory` (
	`VetId`  int(11) NOT NULL,
	`DateId` int(11) NOT NULL,
	`HistDetail` varchar(150) DEFAULT NULL,
	PRIMARY KEY (`VetId`,`DateId`)
);

CREATE TABLE `Pinv` (
	VetId   int(11)     not null,
	PrdId   int(8)      not null,
	InvFch  date        not null,
	InvCnt  int(11)     not null,
	InvSts  varchar(3)  not null,
	InvObs  varchar(70) default null,
	InvUxE  varchar(1)  not null,
	Invdir  varchar(3)  not null,
	PRIMARY KEY(VetId,PrdId,InvFch)
);

CREATE TABLE `Product` (
	VetId   int(11) not null,
	PrdId   int(8)  not null,
	PrdName varchar(30) not null,
	PrdDsc  varchar(70) not null,
	PrdUxE  int(5)  not null,
	PRIMARY KEY (VetId,PrdId)
);

CREATE TABLE `Stock` (
	VetId   int(11) not null,
	PrdId   int(8)  not null,
	StkCnt  int(11) not null,
	PRIMARY KEY (VetId,PrdId)
);
